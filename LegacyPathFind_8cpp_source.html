31 {

39  auto const& jobCount = app.getJobQueue().getJobCountGE(jtCLIENT);
40  if (jobCount > Tuning::maxPathfindJobCount ||
41  app.getFeeTrack().isLoadedLocal())

44  while (true)
45  {
46  int prevVal = inProgress.load();

51  prevVal,
