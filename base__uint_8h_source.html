181  // Helper function to initialize a base_uint from a std::string_view.
182  enum class ParseResult {
183  okay,
184  badLength,
185  badChar,
186  };
187 
188  constexpr std::pair<ParseResult, decltype(data_)>
189  parseFromStringView(std::string_view sv) noexcept
190  {
191  // Local lambda that converts a single hex char to four bits and
192  // ORs those bits into a uint32_t.
193  auto hexCharToUInt = [](char c,
194  std::uint32_t shift,
195  std::uint32_t& accum) -> ParseResult {
196  std::uint32_t nibble = 0xFFu;
197  if (c < '0' || c > 'f')
198  return ParseResult::badChar;
199 
200  if (c >= 'a')
201  nibble = static_cast<std::uint32_t>(c - 'a' + 0xA);
202  else if (c >= 'A')
203  nibble = static_cast<std::uint32_t>(c - 'A' + 0xA);
204  else if (c <= '9')
205  nibble = static_cast<std::uint32_t>(c - '0');
206 
207  if (nibble > 0xFu)
208  return ParseResult::badChar;
209 
210  accum |= (nibble << shift);
211 
212  return ParseResult::okay;
213  };
214 
215  std::pair<ParseResult, decltype(data_)> ret{ParseResult::okay, {}};
216 
217  if (sv == "0")
218  {
219  return ret;
220  }
221 
222  if (sv.size() != size() * 2)
223  {
224  ret.first = ParseResult::badLength;
225  return ret;
226  }
227 
228  auto out = ret.second.data();
229 
230  auto in = sv.begin();
231 
232  while (in != sv.end())
233  {
234  std::uint32_t accum = {};
235  for (std::uint32_t shift : {4u, 0u, 12u, 8u, 20u, 16u, 28u, 24u})
236  {
237  if (auto const result = hexCharToUInt(*in++, shift, accum);
238  result != ParseResult::okay)
239  {
240  ret.first = result;
241  return ret;
242  }
243  }
244  *out++ = accum;
245  }
246  return ret;
247  }
248 
249  constexpr decltype(data_)
250  parseFromStringViewThrows(std::string_view sv) noexcept(false)
251  {
252  auto const result = parseFromStringView(sv);
253  if (result.first == ParseResult::badLength)
254  Throw<std::invalid_argument>("invalid length for hex string");
255 
256  if (result.first == ParseResult::badChar)
257  Throw<std::range_error>("invalid hex character");
258 
259  return result.second;
260  }

437  base_uint&
438  operator+=(const base_uint& b)
439  {
440  std::uint64_t carry = 0;
441 
442  for (int i = WIDTH; i--;)
443  {
444  std::uint64_t n = carry + boost::endian::big_to_native(data_[i]) +
445  boost::endian::big_to_native(b.data_[i]);
446 
447  data_[i] =
448  boost::endian::native_to_big(static_cast<std::uint32_t>(n));
449  carry = n >> 32;
450  }
451 
452  return *this;
453  }

528 template <std::size_t Bits, class Tag>
529 inline int
530 compare(base_uint<Bits, Tag> const& a, base_uint<Bits, Tag> const& b)
531 {
532  auto ret = std::mismatch(a.cbegin(), a.cend(), b.cbegin());
533 
534  if (ret.first == a.cend())
535  return 0;
536 
537  // a > b
538  if (*ret.first > *ret.second)
539  return 1;
540 
541  // a < b
542  return -1;
543 }
