50 std::optional<Manifest>
51 deserializeManifest(Slice s)
52 {
53  if (s.empty())
54  return std::nullopt;
55 
56  static SOTemplate const manifestFormat{
57  // A manifest must include:
58  // - the master public key
59  {sfPublicKey, soeREQUIRED},
60 
61  // - a signature with that public key
62  {sfMasterSignature, soeREQUIRED},
63 
64  // - a sequence number
65  {sfSequence, soeREQUIRED},
66 
67  // It may, optionally, contain:
68  // - a version number which defaults to 0
69  {sfVersion, soeDEFAULT},
70 
71  // - a domain name
72  {sfDomain, soeOPTIONAL},
73 
74  // - an ephemeral signing key that can be changed as necessary
75  {sfSigningPubKey, soeOPTIONAL},
76 
77  // - a signature using the ephemeral signing key, if it is present
78  {sfSignature, soeOPTIONAL},
79  };
80 
81  try
82  {
83  SerialIter sit{s};
84  STObject st{sit, sfGeneric};
85 
86  st.applyTemplate(manifestFormat);
87 
88  // We only understand "version 0" manifests at this time:
89  if (st.isFieldPresent(sfVersion) && st.getFieldU16(sfVersion) != 0)
90  return std::nullopt;
91 
92  auto const pk = st.getFieldVL(sfPublicKey);
93 
94  if (!publicKeyType(makeSlice(pk)))
95  return std::nullopt;
96 
97  Manifest m;
98  m.serialized.assign(reinterpret_cast<char const*>(s.data()), s.size());
99  m.masterKey = PublicKey(makeSlice(pk));
100  m.sequence = st.getFieldU32(sfSequence);
101 
102  if (st.isFieldPresent(sfDomain))
103  {
104  auto const d = st.getFieldVL(sfDomain);
105 
106  m.domain.assign(reinterpret_cast<char const*>(d.data()), d.size());
107 
108  if (!isProperlyFormedTomlDomain(m.domain))
109  return std::nullopt;
110  }
111 
112  bool const hasEphemeralKey = st.isFieldPresent(sfSigningPubKey);
113  bool const hasEphemeralSig = st.isFieldPresent(sfSignature);
114 
115  if (m.revoked())
116  {
117  // Revocation manifests should not specify a new signing key
118  // or a signing key signature.
119  if (hasEphemeralKey)
120  return std::nullopt;
121 
122  if (hasEphemeralSig)
123  return std::nullopt;
124  }
125  else
126  {
127  // Regular manifests should contain a signing key and an
128  // associated signature.
129  if (!hasEphemeralKey)
130  return std::nullopt;
131 
132  if (!hasEphemeralSig)
133  return std::nullopt;
134 
135  auto const spk = st.getFieldVL(sfSigningPubKey);
136 
137  if (!publicKeyType(makeSlice(spk)))
138  return std::nullopt;
139 
140  m.signingKey = PublicKey(makeSlice(spk));
141 
142  // The signing and master keys can't be the same
143  if (m.signingKey == m.masterKey)
144  return std::nullopt;
145  }
146 
147  return m;
148  }
149  catch (std::exception const&)
150  {
151  return std::nullopt;
152  }
153 }

191  // Signing key and signature are not required for

239 std::optional<ValidatorToken>
240 loadValidatorToken(std::vector<std::string> const& blob)
241 {
242  try
243  {
244  std::string tokenStr;
245 
246  tokenStr.reserve(std::accumulate(
247  blob.cbegin(),
248  blob.cend(),
249  std::size_t(0),
250  [](std::size_t init, std::string const& s) {
251  return init + s.size();
252  }));
253 
254  for (auto const& line : blob)
255  tokenStr += boost::algorithm::trim_copy(line);
256 
257  tokenStr = base64_decode(tokenStr);
258 
259  Json::Reader r;
260  Json::Value token;
261 
262  if (r.parse(tokenStr, token))
263  {
264  auto const m = token.get("manifest", Json::Value{});
265  auto const k = token.get("validation_secret_key", Json::Value{});
266 
267  if (m.isString() && k.isString())
268  {
269  auto const key = strUnHex(k.asString());
270 
271  if (key && key->size() == 32)
272  return ValidatorToken{m.asString(), makeSlice(*key)};
273  }
274  }
275 
276  return std::nullopt;
277  }
278  catch (std::exception const&)
279  {
280  return std::nullopt;
281  }
282 }

356 ManifestDisposition
357 ManifestCache::applyManifest(Manifest m)
358 {
359  std::lock_guard applyLock{apply_mutex_};
360 
361  // Before we spend time checking the signature, make sure the
362  // sequence number is newer than any we have.
363  auto const iter = map_.find(m.masterKey);
364 
365  if (iter != map_.end() && m.sequence <= iter->second.sequence)
366  {
367  // We received a manifest whose sequence number is not strictly greater
368  // than the one we already know about. This can happen in several cases
369  // including when we receive manifests from a peer who doesn't have the
370  // latest data.
371  if (auto stream = j_.debug())
372  logMftAct(
373  stream,
374  "Stale",
375  m.masterKey,
376  m.sequence,
377  iter->second.sequence);
378  return ManifestDisposition::stale;
379  }
380 
381  // Now check the signature
382  if (!m.verify())
383  {
384  if (auto stream = j_.warn())
385  logMftAct(stream, "Invalid", m.masterKey, m.sequence);
386  return ManifestDisposition::invalid;
387  }
388 
389  // If the master key associated with a manifest is or might be compromised
390  // and is, therefore, no longer trustworthy.
391  //
392  // A manifest revocation essentially marks a manifest as compromised. By
393  // setting the sequence number to the highest value possible, the manifest
394  // is effectively neutered and cannot be superseded by a forged one.
395  bool const revoked = m.revoked();
396 
397  if (auto stream = j_.warn(); stream && revoked)
398  logMftAct(stream, "Revoked", m.masterKey, m.sequence);
399 
400  std::lock_guard readLock{read_mutex_};
401 
402  // Sanity check: the master key of this manifest should not be used as
403  // the ephemeral key of another manifest:
404  if (auto const x = signingToMasterKeys_.find(m.masterKey);
405  x != signingToMasterKeys_.end())
406  {
407  JLOG(j_.warn()) << to_string(m)
408  << ": Master key already used as ephemeral key for "
409  << toBase58(TokenType::NodePublic, x->second);
410 
411  return ManifestDisposition::badMasterKey;
412  }
413 
414  if (!revoked)
415  {
416  // Sanity check: the ephemeral key of this manifest should not be used
417  // as the master or ephemeral key of another manifest:
418  if (auto const x = signingToMasterKeys_.find(m.signingKey);
419  x != signingToMasterKeys_.end())
420  {
421  JLOG(j_.warn())
422  << to_string(m)
423  << ": Ephemeral key already used as ephemeral key for "
424  << toBase58(TokenType::NodePublic, x->second);
425 
426  return ManifestDisposition::badEphemeralKey;
427  }
428 
429  if (auto const x = map_.find(m.signingKey); x != map_.end())
430  {
431  JLOG(j_.warn())
432  << to_string(m) << ": Ephemeral key used as master key for "
433  << to_string(x->second);
434 
435  return ManifestDisposition::badEphemeralKey;
436  }
437  }
438 
439  // This is the first manifest we are seeing for a master key. This should
440  // only ever happen once per validator run.
441  if (iter == map_.end())
442  {
443  if (auto stream = j_.info())
444  logMftAct(stream, "AcceptedNew", m.masterKey, m.sequence);
445 
446  if (!revoked)
447  signingToMasterKeys_[m.signingKey] = m.masterKey;
448 
449  auto masterKey = m.masterKey;
450  map_.emplace(std::move(masterKey), std::move(m));
451  return ManifestDisposition::accepted;
452  }
453 
454  // An ephemeral key was revoked and superseded by a new key. This is
455  // expected, but should happen infrequently.
456  if (auto stream = j_.info())
457  logMftAct(
458  stream,
459  "AcceptedUpdate",
460  m.masterKey,
461  m.sequence,
462  iter->second.sequence);
463 
464  signingToMasterKeys_.erase(iter->second.signingKey);
465 
466  if (!revoked)
467  signingToMasterKeys_[m.signingKey] = m.masterKey;
468 
469  iter->second = std::move(m);
470 
471  // Something has changed. Keep track of it.
472  seq_++;
473 
474  return ManifestDisposition::accepted;
475 }

484 bool
485 ManifestCache::load(
486  DatabaseCon& dbCon,
487  std::string const& dbTable,
488  std::string const& configManifest,
489  std::vector<std::string> const& configRevocation)
490 {
491  load(dbCon, dbTable);
492 
493  if (!configManifest.empty())
494  {
495  auto mo = deserializeManifest(base64_decode(configManifest));
496  if (!mo)
497  {
498  JLOG(j_.error()) << "Malformed validator_token in config";
499  return false;
500  }
501 
502  if (mo->revoked())
503  {
504  JLOG(j_.warn()) << "Configured manifest revokes public key";
505  }
506 
507  if (applyManifest(std::move(*mo)) == ManifestDisposition::invalid)
508  {
509  JLOG(j_.error()) << "Manifest in config was rejected";
510  return false;
511  }
512  }
513 
514  if (!configRevocation.empty())
515  {
516  std::string revocationStr;
517  revocationStr.reserve(std::accumulate(
518  configRevocation.cbegin(),
519  configRevocation.cend(),
520  std::size_t(0),
521  [](std::size_t init, std::string const& s) {
522  return init + s.size();
523  }));
524 
525  for (auto const& line : configRevocation)
526  revocationStr += boost::algorithm::trim_copy(line);
527 
528  auto mo = deserializeManifest(base64_decode(revocationStr));
529 
530  if (!mo || !mo->revoked() ||
531  applyManifest(std::move(*mo)) == ManifestDisposition::invalid)
532  {
533  JLOG(j_.error()) << "Invalid validator key revocation in config";
534  return false;
535  }
536  }
537 
538  return true;
539 }
