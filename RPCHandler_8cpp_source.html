127 error_code_i
128 fillHandler(JsonContext& context, Handler const*& result)
129 {
130  if (!isUnlimited(context.role))
131  {
132  // VFALCO NOTE Should we also add up the jtRPC jobs?
133  //
134  int jc = context.app.getJobQueue().getJobCountGE(jtCLIENT);
135  if (jc > Tuning::maxJobQueueClients)
136  {
137  JLOG(context.j.debug()) << "Too busy for command: " << jc;
138  return rpcTOO_BUSY;
139  }
140  }
141 
142  if (!context.params.isMember(jss::command) &&
143  !context.params.isMember(jss::method))
144  return rpcCOMMAND_MISSING;
145  if (context.params.isMember(jss::command) &&
146  context.params.isMember(jss::method))
147  {
148  if (context.params[jss::command].asString() !=
149  context.params[jss::method].asString())
150  return rpcUNKNOWN_COMMAND;
151  }
152 
153  std::string strCommand = context.params.isMember(jss::command)
154  ? context.params[jss::command].asString()
155  : context.params[jss::method].asString();
156 
157  JLOG(context.j.trace()) << "COMMAND:" << strCommand;
158  JLOG(context.j.trace()) << "REQUEST:" << context.params;
159  auto handler = getHandler(
160  context.apiVersion, context.app.config().BETA_RPC_API, strCommand);
161 
162  if (!handler)
163  return rpcUNKNOWN_COMMAND;
164 
165  if (handler->role_ == Role::ADMIN && context.role != Role::ADMIN)
166  return rpcNO_PERMISSION;
167 
168  error_code_i res = conditionMet(handler->condition_, context);
169  if (res != rpcSUCCESS)
170  {
171  return res;
172  }
173 
174  result = handler;
175  return rpcSUCCESS;
176 }

178 template <class Object, class Method>
179 Status
180 callMethod(
181  JsonContext& context,
182  Method method,
183  std::string const& name,
184  Object& result)
185 {
186  static std::atomic<std::uint64_t> requestId{0};
187  auto& perfLog = context.app.getPerfLog();
188  std::uint64_t const curId = ++requestId;
189  try
190  {
191  perfLog.rpcStart(name, curId);
192  auto v =
193  context.app.getJobQueue().makeLoadEvent(jtGENERIC, "cmd:" + name);
194 
195  auto start = std::chrono::system_clock::now();
196  auto ret = method(context, result);
197  auto end = std::chrono::system_clock::now();
198 
199  JLOG(context.j.debug())
200  << "RPC call " << name << " completed in "
201  << ((end - start).count() / 1000000000.0) << "seconds";
202  perfLog.rpcFinish(name, curId);
203  return ret;
204  }
205  catch (ReportingShouldProxy&)
206  {
207  result = forwardToP2p(context);
208  return rpcSUCCESS;
209  }
210  catch (std::exception& e)
211  {
212  perfLog.rpcError(name, curId);
213  JLOG(context.j.info()) << "Caught throw: " << e.what();
214 
215  if (context.loadType == Resource::feeReferenceRPC)
216  context.loadType = Resource::feeExceptionRPC;
217 
218  inject_error(rpcINTERNAL, result);
219  return rpcINTERNAL;
220  }
221 }

245 Status
246 doCommand(RPC::JsonContext& context, Json::Value& result)
247 {
248  if (shouldForwardToP2p(context))
249  {
250  result = forwardToP2p(context);
251  injectReportingWarning(context, result);
252  // this return value is ignored
253  return rpcSUCCESS;
254  }
255  Handler const* handler = nullptr;
256  if (auto error = fillHandler(context, handler))
257  {
258  inject_error(error, result);
259  return error;
260  }
261 
262  if (auto method = handler->valueMethod_)
263  {
264  if (!context.headers.user.empty() ||
265  !context.headers.forwardedFor.empty())
266  {
267  JLOG(context.j.debug())
268  << "start command: " << handler->name_
269  << ", user: " << context.headers.user
270  << ", forwarded for: " << context.headers.forwardedFor;
271 
272  auto ret = callMethod(context, method, handler->name_, result);
273 
274  JLOG(context.j.debug())
275  << "finish command: " << handler->name_
276  << ", user: " << context.headers.user
277  << ", forwarded for: " << context.headers.forwardedFor;
278 
279  return ret;
280  }
281  else
282  {
283  auto ret = callMethod(context, method, handler->name_, result);
284  injectReportingWarning(context, result);
285  return ret;
286  }
287  }
288 
289  return rpcUNKNOWN_COMMAND;
290 }
