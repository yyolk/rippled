54  readThreads_.emplace_back(&Database::threadEntry, this);
55 }

88 void
89 Database::stop()
90 {
91  // After stop time we can no longer use the JobQueue for background
92  // reads. Join the background read threads.
93  {
94  std::lock_guard lock(readLock_);
95  if (readStopping_) // Only stop threads once.
96  return;
97 
98  readStopping_ = true;
99  readCondVar_.notify_all();
100  }
101 
102  for (auto& e : readThreads_)
103  e.join();
104 }

118 void
119 Database::importInternal(Backend& dstBackend, Database& srcDB)
120 {
121  Batch batch;
122  batch.reserve(batchWritePreallocationSize);
123  auto storeBatch = [&, fname = __func__]() {
124  try
125  {
126  dstBackend.storeBatch(batch);
127  }
128  catch (std::exception const& e)
129  {
130  JLOG(j_.error()) << "Exception caught in function " << fname
131  << ". Error: " << e.what();
132  return;
133  }
134 
135  std::uint64_t sz{0};
136  for (auto const& nodeObject : batch)
137  sz += nodeObject->getData().size();
138  storeStats(batch.size(), sz);
139  batch.clear();
140  };
141 
142  srcDB.for_each([&](std::shared_ptr<NodeObject> nodeObject) {
143  assert(nodeObject);
144  if (!nodeObject) // This should never happen
145  return;
146 
147  batch.emplace_back(std::move(nodeObject));
148  if (batch.size() >= batchWritePreallocationSize)
149  storeBatch();
150  });
151 
152  if (!batch.empty())
153  storeBatch();
154 }

156 // Perform a fetch and report the time it took
157 std::shared_ptr<NodeObject>
158 Database::fetchNodeObject(
159  uint256 const& hash,
160  std::uint32_t ledgerSeq,
161  FetchType fetchType)
162 {
163  FetchReport fetchReport(fetchType);
164 
165  using namespace std::chrono;
166  auto const begin{steady_clock::now()};
167 
168  auto nodeObject{fetchNodeObject(hash, ledgerSeq, fetchReport)};
169  if (nodeObject)
170  {
171  ++fetchHitCount_;
172  fetchSz_ += nodeObject->getData().size();
173  }
174  ++fetchTotalCount_;
175 
176  fetchReport.elapsed =
177  duration_cast<milliseconds>(steady_clock::now() - begin);
178  scheduler_.onFetch(fetchReport);
179  return nodeObject;
180 }

182 bool
183 Database::storeLedger(
184  Ledger const& srcLedger,
185  std::shared_ptr<Backend> dstBackend)
186 {
187  auto fail = [&](std::string const& msg) {
188  JLOG(j_.error()) << "Source ledger sequence " << srcLedger.info().seq
189  << ". " << msg;
190  return false;
191  };
192 
193  if (srcLedger.info().hash.isZero())
194  return fail("Invalid hash");
195  if (srcLedger.info().accountHash.isZero())
196  return fail("Invalid account hash");
197 
198  auto& srcDB = const_cast<Database&>(srcLedger.stateMap().family().db());
199  if (&srcDB == this)
200  return fail("Source and destination databases are the same");
201 
202  Batch batch;
203  batch.reserve(batchWritePreallocationSize);
204  auto storeBatch = [&, fname = __func__]() {
205  std::uint64_t sz{0};
206  for (auto const& nodeObject : batch)
207  sz += nodeObject->getData().size();
208 
209  try
210  {
211  dstBackend->storeBatch(batch);
212  }
213  catch (std::exception const& e)
214  {
215  fail(
216  std::string("Exception caught in function ") + fname +
217  ". Error: " + e.what());
218  return false;
219  }
220 
221  storeStats(batch.size(), sz);
222  batch.clear();
223  return true;
224  };
225 
226  // Store ledger header
227  {
228  Serializer s(sizeof(std::uint32_t) + sizeof(LedgerInfo));
229  s.add32(HashPrefix::ledgerMaster);
230  addRaw(srcLedger.info(), s);
231  auto nObj = NodeObject::createObject(
232  hotLEDGER, std::move(s.modData()), srcLedger.info().hash);
233  batch.emplace_back(std::move(nObj));
234  }
235 
236  bool error = false;
237  auto visit = [&](SHAMapTreeNode& node) {
238  if (!isStopping())
239  {
240  if (auto nodeObject = srcDB.fetchNodeObject(
241  node.getHash().as_uint256(), srcLedger.info().seq))
242  {
243  batch.emplace_back(std::move(nodeObject));
244  if (batch.size() < batchWritePreallocationSize || storeBatch())
245  return true;
246  }
247  }
248 
249  error = true;
250  return false;
251  };
252 
253  // Store the state map
254  if (srcLedger.stateMap().getHash().isNonZero())
255  {
256  if (!srcLedger.stateMap().isValid())
257  return fail("Invalid state map");
258 
259  srcLedger.stateMap().snapShot(false)->visitNodes(visit);
260  if (error)
261  return fail("Failed to store state map");
262  }
263 
264  // Store the transaction map
265  if (srcLedger.info().txHash.isNonZero())
266  {
267  if (!srcLedger.txMap().isValid())
268  return fail("Invalid transaction map");
269 
270  srcLedger.txMap().snapShot(false)->visitNodes(visit);
271  if (error)
272  return fail("Failed to store transaction map");
273  }
274 
275  if (!batch.empty() && !storeBatch())
276  return fail("Failed to store");
277 
278  return true;
279 }

281 // Entry point for async read threads
282 void
283 Database::threadEntry()
284 {
285  beast::setCurrentThreadName("prefetch");
286  while (true)
287  {
288  uint256 lastHash;
289  std::vector<std::pair<
290  std::uint32_t,
291  std::function<void(std::shared_ptr<NodeObject> const&)>>>
292  entry;
293 
294  {
295  std::unique_lock<std::mutex> lock(readLock_);
296  readCondVar_.wait(
297  lock, [this] { return readStopping_ || !read_.empty(); });
298  if (readStopping_)
299  break;
300 
301  // Read in key order to make the back end more efficient
302  auto it = read_.lower_bound(readLastHash_);
303  if (it == read_.end())
304  {
305  // start over from the beginning
306  it = read_.begin();
307  }
308  lastHash = it->first;
309  entry = std::move(it->second);
310  read_.erase(it);
311  readLastHash_ = lastHash;
312  }
313 
314  auto seq = entry[0].first;
315  auto obj = fetchNodeObject(lastHash, seq, FetchType::async);
316 
317  for (auto const& req : entry)
318  {
319  if ((seq == req.first) || isSameDB(req.first, seq))
320  req.second(obj);
321  else
322  req.second(
323  fetchNodeObject(lastHash, req.first, FetchType::async));
324  }
325  }
326 }
