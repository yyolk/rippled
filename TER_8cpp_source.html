180 bool
181 transResultInfo(TER code, std::string& token, std::string& text)
182 {
183  auto& results = detail::transResults();
184 
185  auto const r = results.find(TERtoInt(code));
186 
187  if (r == results.end())
188  return false;
189 
190  token = r->second.first;
191  text = r->second.second;
192  return true;
193 }

196 transToken(TER code)

201  return transResultInfo(code, token, text) ? token : "-";
202 }

213 std::optional<TER>
214 transCode(std::string const& token)
215 {
216  static auto const results = [] {
217  auto& byTer = detail::transResults();
218  auto range = boost::make_iterator_range(byTer.begin(), byTer.end());
219  auto tRange = boost::adaptors::transform(range, [](auto const& r) {
220  return std::make_pair(r.second.first, r.first);
221  });
222  std::unordered_map<std::string, TERUnderlyingType> const byToken(
223  tRange.begin(), tRange.end());
224  return byToken;
225  }();
226 
227  auto const r = results.find(token);
228 
229  if (r == results.end())
230  return std::nullopt;
231 
232  return TER::fromInt(r->second);
233 }
