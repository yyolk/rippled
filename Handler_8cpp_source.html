30 template <typename Function>
31 Handler::Method<Json::Value>
32 byRef(Function const& f)
33 {
34  return [f](JsonContext& context, Json::Value& result) {
35  result = f(context);
36  if (result.type() != Json::objectValue)
37  {
38  assert(false);
39  result = RPC::makeObjectValue(result);
40  }
41 
42  return Status();
43  };
44 }

46 template <class Object, class HandlerImpl>
47 Status
48 handle(JsonContext& context, Object& object)
49 {
50  HandlerImpl handler(context);
51 
52  auto status = handler.check();
53  if (status)
54  status.inject(object);
55  else
56  handler.writeResult(object);
57  return status;
58 };

167 class HandlerTable
168 {
169 private:
170  template <std::size_t N>
171  explicit HandlerTable(const Handler (&entries)[N])
172  {
173  for (std::size_t i = 0; i < N; ++i)
174  {
175  auto const& entry = entries[i];
176  assert(table_.find(entry.name_) == table_.end());
177  table_[entry.name_] = entry;
178  }
179 
180  // This is where the new-style handlers are added.
181  addHandler<LedgerHandler>();
182  addHandler<VersionHandler>();
183  }
184 
185 public:
186  static HandlerTable const&
187  instance()
188  {
189  static HandlerTable const handlerTable(handlerArray);
190  return handlerTable;
191  }
192 
193  Handler const*
194  getHandler(unsigned version, bool betaEnabled, std::string name) const
195  {
196  if (version < RPC::apiMinimumSupportedVersion ||
197  version > (betaEnabled ? RPC::apiBetaVersion
198  : RPC::apiMaximumSupportedVersion))
199  return nullptr;
200  auto i = table_.find(name);
201  return i == table_.end() ? nullptr : &i->second;
202  }
203 
204  std::vector<char const*>
205  getHandlerNames() const
206  {
207  std::vector<char const*> ret;
208  ret.reserve(table_.size());
209  for (auto const& i : table_)
210  ret.push_back(i.second.name_);
211  return ret;
212  }
213 
214 private:
215  std::map<std::string, Handler> table_;
216 
217  template <class HandlerImpl>
218  void
219  addHandler()
220  {
221  assert(table_.find(HandlerImpl::name()) == table_.end());
222 
223  Handler h;
224  h.name_ = HandlerImpl::name();
225  h.valueMethod_ = &handle<Json::Value, HandlerImpl>;
226  h.role_ = HandlerImpl::role();
227  h.condition_ = HandlerImpl::condition();
228 
229  table_[HandlerImpl::name()] = h;
230  }
231 };
