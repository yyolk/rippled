92 bool
93 getAccountObjects(
94  ReadView const& ledger,
95  AccountID const& account,
96  std::optional<std::vector<LedgerEntryType>> const& typeFilter,
97  uint256 dirIndex,
98  uint256 const& entryIndex,
99  std::uint32_t const limit,
100  Json::Value& jvResult)
101 {
102  auto const root = keylet::ownerDir(account);
103  auto found = false;
104 
105  if (dirIndex.isZero())
106  {
107  dirIndex = root.key;
108  found = true;
109  }
110 
111  auto dir = ledger.read({ltDIR_NODE, dirIndex});
112  if (!dir)
113  return false;
114 
115  std::uint32_t i = 0;
116  auto& jvObjects = (jvResult[jss::account_objects] = Json::arrayValue);
117  for (;;)
118  {
119  auto const& entries = dir->getFieldV256(sfIndexes);
120  auto iter = entries.begin();
121 
122  if (!found)
123  {
124  iter = std::find(iter, entries.end(), entryIndex);
125  if (iter == entries.end())
126  return false;
127 
128  found = true;
129  }
130 
131  for (; iter != entries.end(); ++iter)
132  {
133  auto const sleNode = ledger.read(keylet::child(*iter));
134 
135  auto typeMatchesFilter =
136  [](std::vector<LedgerEntryType> const& typeFilter,
137  LedgerEntryType ledgerType) {
138  auto it = std::find(
139  typeFilter.begin(), typeFilter.end(), ledgerType);
140  return it != typeFilter.end();
141  };
142 
143  if (!typeFilter.has_value() ||
144  typeMatchesFilter(typeFilter.value(), sleNode->getType()))
145  {
146  jvObjects.append(sleNode->getJson(JsonOptions::none));
147 
148  if (++i == limit)
149  {
150  if (++iter != entries.end())
151  {
152  jvResult[jss::limit] = limit;
153  jvResult[jss::marker] =
154  to_string(dirIndex) + ',' + to_string(*iter);
155  return true;
156  }
157 
158  break;
159  }
160  }
161  }
162 
163  auto const nodeIndex = dir->getFieldU64(sfIndexNext);
164  if (nodeIndex == 0)
165  return true;
166 
167  dirIndex = keylet::page(root, nodeIndex).key;
168  dir = ledger.read({ltDIR_NODE, dirIndex});
169  if (!dir)
170  return true;
171 
172  if (i == limit)
173  {
174  auto const& e = dir->getFieldV256(sfIndexes);
175  if (!e.empty())
176  {
177  jvResult[jss::limit] = limit;
178  jvResult[jss::marker] =
179  to_string(dirIndex) + ',' + to_string(*e.begin());
180  }
181 
182  return true;
183  }
184  }

198 template <class T>
199 Status
200 ledgerFromRequest(T& ledger, JsonContext& context)
201 {
202  ledger.reset();
203 
204  auto& params = context.params;
205 
206  auto indexValue = params[jss::ledger_index];
207  auto hashValue = params[jss::ledger_hash];
208 
209  // We need to support the legacy "ledger" field.
210  auto& legacyLedger = params[jss::ledger];
211  if (legacyLedger)
212  {
213  if (legacyLedger.asString().size() > 12)
214  hashValue = legacyLedger;
215  else
216  indexValue = legacyLedger;
217  }
218 
219  if (hashValue)
220  {
221  if (!hashValue.isString())
222  return {rpcINVALID_PARAMS, "ledgerHashNotString"};
223 
224  uint256 ledgerHash;
225  if (!ledgerHash.parseHex(hashValue.asString()))
226  return {rpcINVALID_PARAMS, "ledgerHashMalformed"};
227  return getLedger(ledger, ledgerHash, context);
228  }
229 
230  auto const index = indexValue.asString();
231 
232  if (index == "current" ||
233  (index.empty() && !context.app.config().reporting()))
234  return getLedger(ledger, LedgerShortcut::CURRENT, context);
235 
236  if (index == "validated" ||
237  (index.empty() && context.app.config().reporting()))
238  return getLedger(ledger, LedgerShortcut::VALIDATED, context);
239 
240  if (index == "closed")
241  return getLedger(ledger, LedgerShortcut::CLOSED, context);
242 
243  std::uint32_t iVal;
244  if (beast::lexicalCastChecked(iVal, index))
245  return getLedger(ledger, iVal, context);
246 
247  return {rpcINVALID_PARAMS, "ledgerIndexMalformed"};
248 }

283 template <class T>
284 Status
285 ledgerFromSpecifier(
286  T& ledger,
287  org::xrpl::rpc::v1::LedgerSpecifier const& specifier,
288  Context& context)
289 {
290  ledger.reset();
291 
292  using LedgerCase = org::xrpl::rpc::v1::LedgerSpecifier::LedgerCase;
293  LedgerCase ledgerCase = specifier.ledger_case();
294  switch (ledgerCase)
295  {
296  case LedgerCase::kHash: {
297  uint256 ledgerHash = uint256::fromVoid(specifier.hash().data());
298  return getLedger(ledger, ledgerHash, context);
299  }
300  case LedgerCase::kSequence:
301  return getLedger(ledger, specifier.sequence(), context);
302  case LedgerCase::kShortcut:
303  [[fallthrough]];
304  case LedgerCase::LEDGER_NOT_SET: {
305  auto const shortcut = specifier.shortcut();
306  // note, unspecified defaults to validated in reporting mode
307  if (shortcut ==
308  org::xrpl::rpc::v1::LedgerSpecifier::SHORTCUT_VALIDATED ||
309  (shortcut ==
310  org::xrpl::rpc::v1::LedgerSpecifier::
311  SHORTCUT_UNSPECIFIED &&
312  context.app.config().reporting()))
313  {
314  return getLedger(ledger, LedgerShortcut::VALIDATED, context);
315  }
316  else
317  {
318  if (shortcut ==
319  org::xrpl::rpc::v1::LedgerSpecifier::SHORTCUT_CURRENT ||
320  shortcut ==
321  org::xrpl::rpc::v1::LedgerSpecifier::
322  SHORTCUT_UNSPECIFIED)
323  {
324  return getLedger(ledger, LedgerShortcut::CURRENT, context);
325  }
326  else if (
327  shortcut ==
328  org::xrpl::rpc::v1::LedgerSpecifier::SHORTCUT_CLOSED)
329  {
330  return getLedger(ledger, LedgerShortcut::CLOSED, context);
331  }
332  }
333  }
334  }
335 
336  return Status::OK;
337 }

349 template <class T>
350 Status
351 getLedger(T& ledger, uint32_t ledgerIndex, Context& context)
352 {
353  ledger = context.ledgerMaster.getLedgerBySeq(ledgerIndex);
354  if (ledger == nullptr)
355  {
356  if (context.app.config().reporting())
357  return {rpcLGR_NOT_FOUND, "ledgerNotFound"};
358  auto cur = context.ledgerMaster.getCurrentLedger();
359  if (cur->info().seq == ledgerIndex)
360  {
361  ledger = cur;
362  }
363  }
364 
365  if (ledger == nullptr)
366  return {rpcLGR_NOT_FOUND, "ledgerNotFound"};
367 
368  if (ledger->info().seq > context.ledgerMaster.getValidLedgerIndex() &&
369  isValidatedOld(context.ledgerMaster, context.app.config().standalone()))
370  {
371  ledger.reset();
372  if (context.apiVersion == 1)
373  return {rpcNO_NETWORK, "InsufficientNetworkMode"};
374  return {rpcNOT_SYNCED, "notSynced"};
375  }
376 
377  return Status::OK;
378 }

380 template <class T>
381 Status
382 getLedger(T& ledger, LedgerShortcut shortcut, Context& context)
383 {
384  if (isValidatedOld(
385  context.ledgerMaster,
386  context.app.config().standalone() ||
387  context.app.config().reporting()))
388  {
389  if (context.apiVersion == 1)
390  return {rpcNO_NETWORK, "InsufficientNetworkMode"};
391  return {rpcNOT_SYNCED, "notSynced"};
392  }
393 
394  if (shortcut == LedgerShortcut::VALIDATED)
395  {
396  ledger = context.ledgerMaster.getValidatedLedger();
397  if (ledger == nullptr)
398  {
399  if (context.apiVersion == 1)
400  return {rpcNO_NETWORK, "InsufficientNetworkMode"};
401  return {rpcNOT_SYNCED, "notSynced"};
402  }
403 
404  assert(!ledger->open());
405  }
406  else
407  {
408  if (shortcut == LedgerShortcut::CURRENT)
409  {
410  if (context.app.config().reporting())
411  return {
412  rpcLGR_NOT_FOUND,
413  "Reporting does not track current ledger"};
414  ledger = context.ledgerMaster.getCurrentLedger();
415  assert(ledger->open());
416  }
417  else if (shortcut == LedgerShortcut::CLOSED)
418  {
419  if (context.app.config().reporting())
420  return {
421  rpcLGR_NOT_FOUND, "Reporting does not track closed ledger"};
422  ledger = context.ledgerMaster.getClosedLedger();
423  assert(!ledger->open());
424  }
425  else
426  {
427  return {rpcINVALID_PARAMS, "ledgerIndexMalformed"};
428  }
429 
430  if (ledger == nullptr)
431  {
432  if (context.apiVersion == 1)
433  return {rpcNO_NETWORK, "InsufficientNetworkMode"};
434  return {rpcNOT_SYNCED, "notSynced"};
435  }
436 
437  static auto const minSequenceGap = 10;
438 
439  if (ledger->info().seq + minSequenceGap <
440  context.ledgerMaster.getValidLedgerIndex())
441  {
442  ledger.reset();
443  if (context.apiVersion == 1)
444  return {rpcNO_NETWORK, "InsufficientNetworkMode"};
445  return {rpcNOT_SYNCED, "notSynced"};
446  }
447  }
448  return Status::OK;
449 }

464 bool
465 isValidated(
466  LedgerMaster& ledgerMaster,
467  ReadView const& ledger,
468  Application& app)
469 {
470  if (app.config().reporting())
471  return true;
472 
473  if (ledger.open())
474  return false;
475 
476  if (ledger.info().validated)
477  return true;
478 
479  auto seq = ledger.info().seq;
480  try
481  {
482  // Use the skip list in the last validated ledger to see if ledger
483  // comes before the last validated ledger (and thus has been
484  // validated).
485  auto hash =
486  ledgerMaster.walkHashBySeq(seq, InboundLedger::Reason::GENERIC);
487 
488  if (!hash || ledger.info().hash != *hash)
489  {
490  // This ledger's hash is not the hash of the validated ledger
491  if (hash)
492  {
493  assert(hash->isNonZero());
494  uint256 valHash =
495  app.getRelationalDBInterface().getHashByIndex(seq);
496  if (valHash == ledger.info().hash)
497  {
498  // SQL database doesn't match ledger chain
499  ledgerMaster.clearLedger(seq);
500  }
501  }
502  return false;
503  }
504  }
505  catch (SHAMapMissingNode const& mn)
506  {
507  auto stream = app.journal("RPCHandler").warn();
508  JLOG(stream) << "Ledger #" << seq << ": " << mn.what();
509  return false;
510  }
511 
512  // Mark ledger as validated to save time if we see it again.
513  ledger.info().validated = true;
514  return true;
515 }

517 // The previous version of the lookupLedger command would accept the
518 // "ledger_index" argument as a string and silently treat it as a request to
519 // return the current ledger which, while not strictly wrong, could cause a lot
520 // of confusion.
521 //
522 // The code now robustly validates the input and ensures that the only possible
523 // values for the "ledger_index" parameter are the index of a ledger passed as
524 // an integer or one of the strings "current", "closed" or "validated".
525 // Additionally, the code ensures that the value passed in "ledger_hash" is a
526 // string and a valid hash. Invalid values will return an appropriate error
527 // code.
528 //
529 // In the absence of the "ledger_hash" or "ledger_index" parameters, the code
530 // assumes that "ledger_index" has the value "current".
531 //
532 // Returns a Json::objectValue. If there was an error, it will be in that
533 // return value. Otherwise, the object contains the field "validated" and
534 // optionally the fields "ledger_hash", "ledger_index" and
535 // "ledger_current_index", if they are defined.
536 Status
537 lookupLedger(
538  std::shared_ptr<ReadView const>& ledger,
539  JsonContext& context,
540  Json::Value& result)
541 {
542  if (auto status = ledgerFromRequest(ledger, context))
543  return status;
544 
545  auto& info = ledger->info();
546 
547  if (!ledger->open())
548  {
549  result[jss::ledger_hash] = to_string(info.hash);
550  result[jss::ledger_index] = info.seq;
551  }
552  else
553  {
554  result[jss::ledger_current_index] = info.seq;
555  }
556 
557  result[jss::validated] =
558  isValidated(context.ledgerMaster, *ledger, context.app);
559  return Status::OK;
560 }

632 std::optional<Seed>
633 parseRippleLibSeed(Json::Value const& value)
634 {
635  // ripple-lib encodes seed used to generate an Ed25519 wallet in a
636  // non-standard way. While rippled never encode seeds that way, we
637  // try to detect such keys to avoid user confusion.
638  if (!value.isString())
639  return std::nullopt;
640 
641  auto const result = decodeBase58Token(value.asString(), TokenType::None);
642 
643  if (result.size() == 18 &&
644  static_cast<std::uint8_t>(result[0]) == std::uint8_t(0xE1) &&
645  static_cast<std::uint8_t>(result[1]) == std::uint8_t(0x4B))
646  return Seed(makeSlice(result.substr(2)));
647 
648  return std::nullopt;
649 }

651 std::optional<Seed>
652 getSeedFromRPC(Json::Value const& params, Json::Value& error)
653 {
654  // The array should be constexpr, but that makes Visual Studio unhappy.
655  static char const* const seedTypes[]{
656  jss::passphrase.c_str(), jss::seed.c_str(), jss::seed_hex.c_str()};
657 
658  // Identify which seed type is in use.
659  char const* seedType = nullptr;
660  int count = 0;
661  for (auto t : seedTypes)
662  {
663  if (params.isMember(t))
664  {
665  ++count;
666  seedType = t;
667  }
668  }
669 
670  if (count != 1)
671  {
672  error = RPC::make_param_error(
673  "Exactly one of the following must be specified: " +
674  std::string(jss::passphrase) + ", " + std::string(jss::seed) +
675  " or " + std::string(jss::seed_hex));
676  return std::nullopt;
677  }
678 
679  // Make sure a string is present
680  if (!params[seedType].isString())
681  {
682  error = RPC::expected_field_error(seedType, "string");
683  return std::nullopt;
684  }
685 
686  auto const fieldContents = params[seedType].asString();
687 
688  // Convert string to seed.
689  std::optional<Seed> seed;
690 
691  if (seedType == jss::seed.c_str())
692  seed = parseBase58<Seed>(fieldContents);
693  else if (seedType == jss::passphrase.c_str())
694  seed = parseGenericSeed(fieldContents);
695  else if (seedType == jss::seed_hex.c_str())
696  {
697  uint128 s;
698 
699  if (s.parseHex(fieldContents))
700  seed.emplace(Slice(s.data(), s.size()));
701  }
702 
703  if (!seed)
704  error = rpcError(rpcBAD_SEED);
705 
706  return seed;
707 }

709 std::pair<PublicKey, SecretKey>
710 keypairForSignature(Json::Value const& params, Json::Value& error)
711 {
712  bool const has_key_type = params.isMember(jss::key_type);
713 
714  // All of the secret types we allow, but only one at a time.
715  // The array should be constexpr, but that makes Visual Studio unhappy.
716  static char const* const secretTypes[]{
717  jss::passphrase.c_str(),
718  jss::secret.c_str(),
719  jss::seed.c_str(),
720  jss::seed_hex.c_str()};
721 
722  // Identify which secret type is in use.
723  char const* secretType = nullptr;
724  int count = 0;
725  for (auto t : secretTypes)
726  {
727  if (params.isMember(t))
728  {
729  ++count;
730  secretType = t;
731  }
732  }
733 
734  if (count == 0 || secretType == nullptr)
735  {
736  error = RPC::missing_field_error(jss::secret);
737  return {};

793  return {};
794  }
795 
796  keyType = KeyType::ed25519;
797  }
798  }

801  keyType = KeyType::secp256k1;
802 
803  if (!seed)

807  else

812  return {};
813  }
814

816  }
817  }

821  if (!contains_error(error))
822  {
823  error = RPC::make_error(

825  }
826 
827  return {};
828  }
829

831  LogicError("keypairForSignature: invalid key type");

834 }
