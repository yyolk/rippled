72 class StreambufWSMsg : public WSMsg
73 {
74  Streambuf sb_;
75  std::size_t n_ = 0;
76 
77 public:
78  StreambufWSMsg(Streambuf&& sb) : sb_(std::move(sb))
79  {
80  }
81 
82  std::pair<boost::tribool, std::vector<boost::asio::const_buffer>>
83  prepare(std::size_t bytes, std::function<void(void)>) override
84  {
85  if (sb_.size() == 0)
86  return {true, {}};
87  sb_.consume(n_);
88  boost::tribool done;
89  if (bytes < sb_.size())
90  {
91  n_ = bytes;
92  done = false;
93  }
94  else
95  {
96  n_ = sb_.size();
97  done = true;
98  }
99  auto const pb = boost::beast::buffers_prefix(n_, sb_.data());
100  std::vector<boost::asio::const_buffer> vb(
101  std::distance(pb.begin(), pb.end()));
102  std::copy(pb.begin(), pb.end(), std::back_inserter(vb));
103  return {done, vb};
104  }
105 };
