35 DeleteAccount::preflight(PreflightContext const& ctx)
36 {
37  if (!ctx.rules.enabled(featureDeletableAccounts))
38  return temDISABLED;
39 
40  if (ctx.tx.getFlags() & tfUniversalMask)
41  return temINVALID_FLAG;
42 
43  auto const ret = preflight1(ctx);
44 
45  if (!isTesSuccess(ret))
46  return ret;
47 
48  if (ctx.tx[sfAccount] == ctx.tx[sfDestination])
49  // An account cannot be deleted and give itself the resulting XRP.
50  return temDST_IS_SRC;
51 
52  return preflight2(ctx);
53 }

56 DeleteAccount::calculateBaseFee(ReadView const& view, STTx const& tx)
57 {
58  // The fee required for AccountDelete is one owner reserve. But the
59  // owner reserve is stored in drops. We need to convert it to fee units.
60  Fees const& fees{view.fees()};
61  std::pair<bool, FeeUnit64> const mulDivResult{
62  mulDiv(fees.increment, safe_cast<FeeUnit64>(fees.units), fees.base)};
63  if (mulDivResult.first)
64  return mulDivResult.second;
65 
66  // If mulDiv returns false then overflow happened. Punt by using the
67  // standard calculation.
68  return Transactor::calculateBaseFee(view, tx);
69 }

154 DeleteAccount::preclaim(PreclaimContext const& ctx)
155 {
156  AccountID const account{ctx.tx[sfAccount]};
157  AccountID const dst{ctx.tx[sfDestination]};
158 
159  auto sleDst = ctx.view.read(keylet::account(dst));
160 
161  if (!sleDst)
162  return tecNO_DST;
163 
164  if ((*sleDst)[sfFlags] & lsfRequireDestTag && !ctx.tx[~sfDestinationTag])
165  return tecDST_TAG_NEEDED;
166 
167  // Check whether the destination account requires deposit authorization.
168  if (ctx.view.rules().enabled(featureDepositAuth) &&
169  (sleDst->getFlags() & lsfDepositAuth))
170  {
171  if (!ctx.view.exists(keylet::depositPreauth(dst, account)))
172  return tecNO_PERMISSION;
173  }
174 
175  auto sleAccount = ctx.view.read(keylet::account(account));
176  assert(sleAccount);
177  if (!sleAccount)
178  return terNO_ACCOUNT;
179 
180  // We don't allow an account to be deleted if its sequence number
181  // is within 256 of the current ledger. This prevents replay of old
182  // transactions if this account is resurrected after it is deleted.
183  //
184  // We look at the account's Sequence rather than the transaction's
185  // Sequence in preparation for Tickets.
186  constexpr std::uint32_t seqDelta{255};
187  if ((*sleAccount)[sfSequence] + seqDelta > ctx.view.seq())
188  return tecTOO_SOON;
189 
190  // Verify that the account does not own any objects that would prevent
191  // the account from being deleted.
192  Keylet const ownerDirKeylet{keylet::ownerDir(account)};
193  if (dirIsEmpty(ctx.view, ownerDirKeylet))
194  return tesSUCCESS;
195 
196  std::shared_ptr<SLE const> sleDirNode{};
197  unsigned int uDirEntry{0};
198  uint256 dirEntry{beast::zero};
199 
200  if (!cdirFirst(
201  ctx.view,
202  ownerDirKeylet.key,
203  sleDirNode,
204  uDirEntry,
205  dirEntry,
206  ctx.j))
207  // Account has no directory at all. This _should_ have been caught
208  // by the dirIsEmpty() check earlier, but it's okay to catch it here.
209  return tesSUCCESS;
210 
211  std::int32_t deletableDirEntryCount{0};
212  do
213  {
214  // Make sure any directory node types that we find are the kind
215  // we can delete.
216  Keylet const itemKeylet{ltCHILD, dirEntry};
217  auto sleItem = ctx.view.read(itemKeylet);
218  if (!sleItem)
219  {
220  // Directory node has an invalid index. Bail out.
221  JLOG(ctx.j.fatal())
222  << "DeleteAccount: directory node in ledger " << ctx.view.seq()
223  << " has index to object that is missing: "
224  << to_string(dirEntry);
225  return tefBAD_LEDGER;
226  }
227 
228  LedgerEntryType const nodeType{
229  safe_cast<LedgerEntryType>((*sleItem)[sfLedgerEntryType])};
230 
231  if (!nonObligationDeleter(nodeType))
232  return tecHAS_OBLIGATIONS;
233 
234  // We found a deletable directory entry. Count it. If we find too
235  // many deletable directory entries then bail out.
236  if (++deletableDirEntryCount > maxDeletableDirEntries)
237  return tefTOO_BIG;
238 
239  } while (cdirNext(
240  ctx.view, ownerDirKeylet.key, sleDirNode, uDirEntry, dirEntry, ctx.j));
241 
242  return tesSUCCESS;
243 }

245 TER
246 DeleteAccount::doApply()
247 {
248  auto src = view().peek(keylet::account(account_));
249  assert(src);
250 
251  auto dst = view().peek(keylet::account(ctx_.tx[sfDestination]));
252  assert(dst);
253 
254  if (!src || !dst)
255  return tefBAD_LEDGER;
256 
257  // Delete all of the entries in the account directory.
258  Keylet const ownerDirKeylet{keylet::ownerDir(account_)};
259  std::shared_ptr<SLE> sleDirNode{};
260  unsigned int uDirEntry{0};
261  uint256 dirEntry{beast::zero};
262 
263  if (view().exists(ownerDirKeylet) &&
264  dirFirst(
265  view(), ownerDirKeylet.key, sleDirNode, uDirEntry, dirEntry, j_))
266  {
267  do
268  {
269  // Choose the right way to delete each directory node.
270  Keylet const itemKeylet{ltCHILD, dirEntry};
271  auto sleItem = view().peek(itemKeylet);
272  if (!sleItem)
273  {
274  // Directory node has an invalid index. Bail out.
275  JLOG(j_.fatal())
276  << "DeleteAccount: Directory node in ledger "
277  << view().seq() << " has index to object that is missing: "
278  << to_string(dirEntry);
279  return tefBAD_LEDGER;
280  }
281 
282  LedgerEntryType const nodeType{safe_cast<LedgerEntryType>(
283  sleItem->getFieldU16(sfLedgerEntryType))};
284 
285  if (auto deleter = nonObligationDeleter(nodeType))
286  {
287  TER const result{
288  deleter(ctx_.app, view(), account_, dirEntry, sleItem, j_)};
289 
290  if (!isTesSuccess(result))
291  return result;
292  }
293  else
294  {
295  assert(!"Undeletable entry should be found in preclaim.");
296  JLOG(j_.error())
297  << "DeleteAccount undeletable item not found in preclaim.";
298  return tecHAS_OBLIGATIONS;
299  }
300 
301  // dirFirst() and dirNext() are like iterators with exposed
302  // internal state. We'll take advantage of that exposed state
303  // to solve a common C++ problem: iterator invalidation while
304  // deleting elements from a container.
305  //
306  // We have just deleted one directory entry, which means our
307  // "iterator state" is invalid.
308  //
309  // 1. During the process of getting an entry from the
310  // directory uDirEntry was incremented from 0 to 1.
311  //
312  // 2. We then deleted the entry at index 0, which means the
313  // entry that was at 1 has now moved to 0.
314  //
315  // 3. So we verify that uDirEntry is indeed 1. Then we jam it
316  // back to zero to "un-invalidate" the iterator.
317  assert(uDirEntry == 1);
318  if (uDirEntry != 1)
319  {
320  JLOG(j_.error())
321  << "DeleteAccount iterator re-validation failed.";
322  return tefBAD_LEDGER;
323  }
324  uDirEntry = 0;
325 
326  } while (dirNext(
327  view(), ownerDirKeylet.key, sleDirNode, uDirEntry, dirEntry, j_));
328  }
329 
330  // Transfer any XRP remaining after the fee is paid to the destination:
331  (*dst)[sfBalance] = (*dst)[sfBalance] + mSourceBalance;
332  (*src)[sfBalance] = (*src)[sfBalance] - mSourceBalance;
333  ctx_.deliver(mSourceBalance);
334 
335  assert((*src)[sfBalance] == XRPAmount(0));
336 
337  // If there's still an owner directory associated with the source account
338  // delete it.
339  if (view().exists(ownerDirKeylet) && !view().emptyDirDelete(ownerDirKeylet))
340  {
341  JLOG(j_.error()) << "DeleteAccount cannot delete root dir node of "
342  << toBase58(account_);
343  return tecHAS_OBLIGATIONS;
344  }
345 
346  // Re-arm the password change fee if we can and need to.
347  if (mSourceBalance > XRPAmount(0) && dst->isFlag(lsfPasswordSpent))
348  dst->clearFlag(lsfPasswordSpent);
349 
350  view().update(dst);
351  view().erase(src);
352 
353  return tesSUCCESS;
354 }
