49 Account::fromCache(std::string name, KeyType type)
50 {
51  auto p = std::make_pair(name, type); // non-const so it can be moved from
52  auto const iter = cache_.find(p);
53  if (iter != cache_.end())
54  return iter->second;
55 
56  auto const keys = generateKeyPair(type, generateSeed(name));
57  auto r = cache_.emplace(
58  std::piecewise_construct,
59  std::forward_as_tuple(std::move(p)),
60  std::forward_as_tuple(std::move(name), keys, privateCtorTag{}));
61  return r.first->second;
62 }
