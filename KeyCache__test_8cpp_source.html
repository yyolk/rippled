27 class KeyCache_test : public beast::unit_test::suite
28 {
29 public:
30  void
31  run() override
32  {
33  using namespace std::chrono_literals;
34  TestStopwatch clock;
35  clock.set(0);
36 
37  using Key = std::string;
38  using Cache = KeyCache<Key>;
39 
40  // Insert an item, retrieve it, and age it so it gets purged.
41  {
42  Cache c("test", clock, 1, 2s);
43 
44  BEAST_EXPECT(c.size() == 0);
45  BEAST_EXPECT(c.insert("one"));
46  BEAST_EXPECT(!c.insert("one"));
47  BEAST_EXPECT(c.size() == 1);
48  BEAST_EXPECT(c.exists("one"));
49  BEAST_EXPECT(c.touch_if_exists("one"));
50  ++clock;
51  c.sweep();
52  BEAST_EXPECT(c.size() == 1);
53  BEAST_EXPECT(c.exists("one"));
54  ++clock;
55  c.sweep();
56  BEAST_EXPECT(c.size() == 0);
57  BEAST_EXPECT(!c.exists("one"));
58  BEAST_EXPECT(!c.touch_if_exists("one"));
59  }
60 
61  // Insert two items, have one expire
62  {
63  Cache c("test", clock, 2, 2s);
64 
65  BEAST_EXPECT(c.insert("one"));
66  BEAST_EXPECT(c.size() == 1);
67  BEAST_EXPECT(c.insert("two"));
68  BEAST_EXPECT(c.size() == 2);
69  ++clock;
70  c.sweep();
71  BEAST_EXPECT(c.size() == 2);
72  BEAST_EXPECT(c.touch_if_exists("two"));
73  ++clock;
74  c.sweep();
75  BEAST_EXPECT(c.size() == 1);
76  BEAST_EXPECT(c.exists("two"));
77  }
78 
79  // Insert three items (1 over limit), sweep
80  {
81  Cache c("test", clock, 2, 3s);
82 
83  BEAST_EXPECT(c.insert("one"));
84  ++clock;
85  BEAST_EXPECT(c.insert("two"));
86  ++clock;
87  BEAST_EXPECT(c.insert("three"));
88  ++clock;
89  BEAST_EXPECT(c.size() == 3);
90  c.sweep();
91  BEAST_EXPECT(c.size() < 3);
92  }
93  }
94 };
