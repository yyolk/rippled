97 template <class Buffers>
98 bool
99 ZeroCopyInputStream<Buffers>::Skip(int count)
100 {
101  if (first_ == last_)
102  return false;
103  while (count > 0)
104  {
105  auto const size = boost::asio::buffer_size(pos_);
106  if (count < size)
107  {
108  pos_ = pos_ + count;
109  count_ += count;
110  return true;
111  }
112  count_ += size;
113  if (++first_ == last_)
114  return false;
115  count -= size;
116  pos_ = *first_;
117  }
118  return true;

180 template <class Streambuf>
181 bool
182 ZeroCopyOutputStream<Streambuf>::Next(void** data, int* size)
183 {
184  if (commit_ != 0)
185  {
186  streambuf_.commit(commit_);
187  count_ += commit_;
188  }
189 
190  if (pos_ == buffers_.end())
191  {
192  buffers_ = streambuf_.prepare(blockSize_);
193  pos_ = buffers_.begin();
194  }
195 
196  *data = boost::asio::buffer_cast<void*>(*pos_);
197  *size = boost::asio::buffer_size(*pos_);
198  commit_ = *size;
199  ++pos_;
200  return true;
201 }
