39 std::unique_ptr<RelationalDBInterface>
40 RelationalDBInterface::init(
41  Application& app,
42  Config const& config,
43  JobQueue& jobQueue)
44 {
45  bool use_sqlite = false;
46  bool use_postgres = false;
47 
48  if (config.reporting())
49  {
50  use_postgres = true;
51  }
52  else
53  {
54  const Section& rdb_section{config.section(SECTION_RELATIONAL_DB)};
55  if (!rdb_section.empty())
56  {
57  if (boost::iequals(
58  get<std::string>(rdb_section, "backend"), "sqlite"))
59  {
60  use_sqlite = true;
61  }
62  else
63  {
64  Throw<std::runtime_error>(
65  "Invalid rdb_section backend value: " +
66  get<std::string>(rdb_section, "backend"));
67  }
68  }
69  else
70  {
71  use_sqlite = true;
72  }
73  }
74 
75  if (use_sqlite)
76  {
77  return getRelationalDBInterfaceSqlite(app, config, jobQueue);
78  }
79  else if (use_postgres)
80  {
81  return getRelationalDBInterfacePostgres(app, config, jobQueue);
82  }
83 
84  return std::unique_ptr<RelationalDBInterface>();
85 }
