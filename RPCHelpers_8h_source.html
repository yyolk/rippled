231 template <class Object>
232 void
233 setVersion(Object& parent, unsigned int apiVersion, bool betaEnabled)
234 {
235  assert(apiVersion != apiInvalidVersion);
236  auto&& object = addObject(parent, jss::version);
237  if (apiVersion == apiVersionIfUnspecified)
238  {
239  object[jss::first] = firstVersion.print();
240  object[jss::good] = goodVersion.print();
241  object[jss::last] = lastVersion.print();
242  }
243  else
244  {
245  object[jss::first] = apiMinimumSupportedVersion;
246  object[jss::last] =
247  betaEnabled ? apiBetaVersion : apiMaximumSupportedVersion;
248  }
249 }
