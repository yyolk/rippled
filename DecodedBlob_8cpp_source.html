28 DecodedBlob::DecodedBlob(void const* key, void const* value, int valueBytes)
29 {
30  /* Data format:
31 
32  Bytes
33 
34  0...7 Unused
35  8 char One of NodeObjectType
36  9...end The body of the object data
37  */
38 
39  m_success = false;
40  m_key = key;
41  // VFALCO NOTE Ledger indexes should have started at 1
42  m_objectType = hotUNKNOWN;
43  m_objectData = nullptr;
44  m_dataBytes = std::max(0, valueBytes - 9);
45 
46  // VFALCO NOTE What about bytes 4 through 7 inclusive?
47 
48  if (valueBytes > 8)
49  {
50  unsigned char const* byte = static_cast<unsigned char const*>(value);
51  m_objectType = safe_cast<NodeObjectType>(byte[8]);
52  }
53 
54  if (valueBytes > 9)
55  {
56  m_objectData = static_cast<unsigned char const*>(value) + 9;
57 
58  switch (m_objectType)
59  {
60  default:
61  break;
62 
63  case hotUNKNOWN:
64  case hotLEDGER:
65  case hotACCOUNT_NODE:
66  case hotTRANSACTION_NODE:
67  m_success = true;
68  break;
69  }
70  }
71 }
