85 std::uint64_t
86 encodeSoftwareVersion(char const* const versionStr)
87 {
88  std::uint64_t c = implementationVersionIdentifier;
89 
90  beast::SemanticVersion v;
91 
92  if (v.parse(std::string(versionStr)))
93  {
94  if (v.majorVersion >= 0 && v.majorVersion <= 255)
95  c |= static_cast<std::uint64_t>(v.majorVersion) << 40;
96 
97  if (v.minorVersion >= 0 && v.minorVersion <= 255)
98  c |= static_cast<std::uint64_t>(v.minorVersion) << 32;
99 
100  if (v.patchVersion >= 0 && v.patchVersion <= 255)
101  c |= static_cast<std::uint64_t>(v.patchVersion) << 24;
102 
103  if (!v.isPreRelease())
104  c |= static_cast<std::uint64_t>(0xC00000);
105 
106  if (v.isPreRelease())
107  {
108  std::uint8_t x = 0;
109 
110  for (auto id : v.preReleaseIdentifiers)
111  {
112  auto parsePreRelease = [](std::string_view identifier,
113  std::string_view prefix,
114  std::uint8_t key,
115  std::uint8_t lok,
116  std::uint8_t hik) -> std::uint8_t {
117  std::uint8_t ret = 0;
118 
119  if (prefix != identifier.substr(0, prefix.length()))
120  return 0;
121 
122  if (!beast::lexicalCastChecked(
123  ret,
124  std::string(identifier.substr(prefix.length()))))
125  return 0;
126 
127  if (std::clamp(ret, lok, hik) != ret)
128  return 0;
129 
130  return ret + key;
131  };
132 
133  x = parsePreRelease(id, "rc", 0x80, 0, 63);
134 
135  if (x == 0)
136  x = parsePreRelease(id, "b", 0x40, 0, 63);
137 
138  if (x & 0xC0)
139  {
140  c |= static_cast<std::uint64_t>(x) << 16;
141  break;
142  }
143  }
144  }
145  }
146 
147  return c;
148 }
