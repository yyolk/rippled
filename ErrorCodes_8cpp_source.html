122 // C++ does not allow you to return an array from a function. You must
123 // return an object which may in turn contain an array. The following
124 // struct is simply defined so the enclosed array can be returned from a
125 // constexpr function.
126 //
127 // In C++17 this struct can be replaced by a std::array. But in C++14
128 // the constexpr methods of a std::array are not sufficient to perform the
129 // necessary work at compile time.
130 template <int N>
131 struct ErrorInfoArray
132 {
133  // Visual Studio doesn't treat a templated aggregate as an aggregate.
134  // So, for Visual Studio, we define a constexpr default constructor.
135  constexpr ErrorInfoArray() : infos{}
136  {
137  }
138 
139  ErrorInfo infos[N];
140 };

142 // Sort and validate unorderedErrorInfos at compile time. Should be
143 // converted to consteval when get to C++20.
144 template <int M, int N>
145 constexpr auto
146 sortErrorInfos(ErrorInfo const (&unordered)[N]) -> ErrorInfoArray<M>
147 {
148  ErrorInfoArray<M> ret;
149 
150  for (ErrorInfo const& info : unordered)
151  {
152  if (info.code <= rpcSUCCESS || info.code > rpcLAST)
153  throw(std::out_of_range("Invalid error_code_i"));
154 
155  // The first valid code follows rpcSUCCESS immediately.
156  static_assert(rpcSUCCESS == 0, "Unexpected error_code_i layout.");
157  int const index{info.code - 1};
158 
159  if (ret.infos[index].code != rpcUNKNOWN)
160  throw(std::invalid_argument("Duplicate error_code_i in list"));
161 
162  ret.infos[index].code = info.code;
163  ret.infos[index].token = info.token;
164  ret.infos[index].message = info.message;
165  }
166 
167  // Verify that all entries are filled in starting with 1 and proceeding
168  // to rpcLAST.
169  //
170  // It's okay for there to be missing entries; they will contain the code
171  // rpcUNKNOWN. But other than that all entries should match their index.
172  int codeCount{0};
173  int expect{rpcBAD_SYNTAX - 1};
174  for (ErrorInfo const& info : ret.infos)
175  {
176  ++expect;
177  if (info.code == rpcUNKNOWN)
178  continue;
179 
180  if (info.code != expect)
181  throw(std::invalid_argument("Empty error_code_i in list"));
182  ++codeCount;
183  }
184  if (expect != rpcLAST)
185  throw(std::invalid_argument("Insufficient list entries"));
186  if (codeCount != N)
187  throw(std::invalid_argument("Bad handling of unorderedErrorInfos"));
188 
189  return ret;
190 }
