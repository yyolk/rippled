36 bool
37 LoadFeeTrack::raiseLocalFee()
38 {
39  std::lock_guard sl(lock_);
40 
41  if (++raiseCount_ < 2)
42  return false;
43 
44  std::uint32_t const origFee = localTxnLoadFee_;
45 
46  // make sure this fee takes effect
47  if (localTxnLoadFee_ < remoteTxnLoadFee_)
48  localTxnLoadFee_ = remoteTxnLoadFee_;
49 
50  // Increase slowly
51  localTxnLoadFee_ += (localTxnLoadFee_ / lftFeeIncFraction);
52 
53  if (localTxnLoadFee_ > lftFeeMax)
54  localTxnLoadFee_ = lftFeeMax;
55 
56  if (origFee == localTxnLoadFee_)
57  return false;
58 
59  JLOG(j_.debug()) << "Local load fee raised from " << origFee << " to "
60  << localTxnLoadFee_;
61  return true;
62 }

69  raiseCount_ = 0;

87 // Scale using load as well as base rate
88 XRPAmount
89 scaleFeeLoad(
90  FeeUnit64 fee,
91  LoadFeeTrack const& feeTrack,
92  Fees const& fees,
93  bool bUnlimited)
94 {
95  if (fee == 0)
96  return XRPAmount{0};
97 
98  // Normally, types with different units wouldn't be mathematically
99  // compatible. This function is an exception.
100  auto lowestTerms = [](auto& a, auto& b) {
101  auto value = [](auto val) {
102  if constexpr (std::is_arithmetic_v<decltype(val)>)
103  return val;
104  else
105  return val.value();
106  };
107 
108  if (auto const g = std::gcd(value(a), value(b)))
109  {
110  a = value(a) / g;
111  b = value(b) / g;
112  }
113  };
114 
115  // Collect the fee rates
116  auto [feeFactor, uRemFee] = feeTrack.getScalingFactors();
117 
118  // Let privileged users pay the normal fee until
119  // the local load exceeds four times the remote.
120  if (bUnlimited && (feeFactor > uRemFee) && (feeFactor < (4 * uRemFee)))
121  feeFactor = uRemFee;
122 
123  XRPAmount baseFee{fees.base};
124  // Compute:
125  // fee = fee * baseFee * feeFactor / (fees.units * lftNormalFee);
126  // without overflow, and as accurately as possible
127 
128  // The denominator of the fraction we're trying to compute.
129  // fees.units and lftNormalFee are both 32 bit,
130  // so the multiplication can't overflow.
131  auto den = FeeUnit64{fees.units} *
132  safe_cast<std::uint64_t>(feeTrack.getLoadBase());
133  // Reduce fee * baseFee * feeFactor / (fees.units * lftNormalFee)
134  // to lowest terms.
135  lowestTerms(fee, den);
136  lowestTerms(baseFee, den);
137  lowestTerms(feeFactor, den);
138 
139  // fee and baseFee are 64 bit, feeFactor is 32 bit
140  // Order fee and baseFee largest first
141  // Normally, these types wouldn't be comparable or swappable.
142  // This function is an exception.
143  if (fee.value() < baseFee.value())
144  {
145  auto tmp = fee.value();
146  fee = baseFee.value();
147  baseFee = tmp;
148  }
149  // double check
150  assert(fee.value() >= baseFee.value());
151 
152  // If baseFee * feeFactor overflows, the final result will overflow
153  XRPAmount const baseFeeOverflow{
154  std::numeric_limits<XRPAmount::value_type>::max() / feeFactor};
155  if (baseFee > baseFeeOverflow)
156  {
157  Throw<std::overflow_error>("scaleFeeLoad");
158  }
159  baseFee *= feeFactor;
160 
161  auto const result = mulDiv(fee, baseFee, den);
162  if (!result.first)
163  Throw<std::overflow_error>("scaleFeeLoad");
164  return result.second;
165 }
