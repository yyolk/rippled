150 void
151 PeerImp::run()
152 {
153  if (!strand_.running_in_this_thread())
154  return post(strand_, std::bind(&PeerImp::run, shared_from_this()));
155 
156  auto parseLedgerHash =
157  [](std::string const& value) -> std::optional<uint256> {
158  if (uint256 ret; ret.parseHex(value))
159  return ret;
160 
161  if (auto const s = base64_decode(value); s.size() == uint256::size())
162  return uint256{s};
163 
164  return std::nullopt;
165  };
166 
167  std::optional<uint256> closed;
168  std::optional<uint256> previous;
169 
170  if (auto const iter = headers_.find("Closed-Ledger");
171  iter != headers_.end())
172  {
173  closed = parseLedgerHash(iter->value().to_string());
174 
175  if (!closed)
176  fail("Malformed handshake data (1)");
177  }
178 
179  if (auto const iter = headers_.find("Previous-Ledger");
180  iter != headers_.end())
181  {
182  previous = parseLedgerHash(iter->value().to_string());
183 
184  if (!previous)
185  fail("Malformed handshake data (2)");
186  }
187 
188  if (previous && !closed)
189  fail("Malformed handshake data (3)");
190 
191  {
192  std::lock_guard<std::mutex> sl(recentLock_);
193  if (closed)
194  closedLedgerHash_ = *closed;
195  if (previous)
196  previousLedgerHash_ = *previous;
197  }
198 
199  if (inbound_)
200  doAccept();
201  else
202  doProtocolStart();
203 
204  // Anything else that needs to be done with the connection should be
205  // done in doProtocolStart
206 }

235 PeerImp::send(std::shared_ptr<Message> const& m)
236 {
237  if (!strand_.running_in_this_thread())
238  return post(strand_, std::bind(&PeerImp::send, shared_from_this(), m));
239  if (gracefulClose_)
240  return;
241  if (detaching_)
242  return;
243 
244  auto validator = m->getValidatorKey();
245  if (validator && !squelch_.expireSquelch(*validator))
246  return;
247 
248  overlay_.reportTraffic(
249  safe_cast<TrafficCount::category>(m->getCategory()),
250  false,
251  static_cast<int>(m->getBuffer(compressionEnabled_).size()));
252 
253  auto sendq_size = send_queue_.size();
254 
255  if (sendq_size < Tuning::targetSendQueue)
256  {
257  // To detect a peer that does not read from their
258  // side of the connection, we expect a peer to have
259  // a small senq periodically
260  large_sendq_ = 0;
261  }
262  else if (auto sink = journal_.debug();
263  sink && (sendq_size % Tuning::sendQueueLogFreq) == 0)
264  {
265  std::string const n = name();
266  sink << (n.empty() ? remote_address_.to_string() : n)
267  << " sendq: " << sendq_size;
268  }
269 
270  send_queue_.push(m);
271 
272  if (sendq_size != 0)
273  return;
274 
275  boost::asio::async_write(
276  stream_,
277  boost::asio::buffer(
278  send_queue_.front()->getBuffer(compressionEnabled_)),
279  bind_executor(
280  strand_,
281  std::bind(
282  &PeerImp::onWriteMessage,
283  shared_from_this(),
284  std::placeholders::_1,
285  std::placeholders::_2)));

416  case protocol::nsVALIDATING:
417  ret[jss::status] = "validating";

640 void
641 PeerImp::onTimer(error_code const& ec)
642 {
643  if (!socket_.is_open())
644  return;
645 
646  if (ec == boost::asio::error::operation_aborted)
647  return;
648 
649  if (ec)
650  {
651  // This should never happen
652  JLOG(journal_.error()) << "onTimer: " << ec.message();
653  return close();
654  }
655 
656  if (large_sendq_++ >= Tuning::sendqIntervals)
657  {
658  fail("Large send queue");
659  return;
660  }
661 
662  if (auto const t = tracking_.load(); !inbound_ && t != Tracking::converged)
663  {
664  clock_type::duration duration;
665 
666  {
667  std::lock_guard sl(recentLock_);
668  duration = clock_type::now() - trackingTime_;
669  }
670 
671  if ((t == Tracking::diverged &&
672  (duration > app_.config().MAX_DIVERGED_TIME)) ||
673  (t == Tracking::unknown &&
674  (duration > app_.config().MAX_UNKNOWN_TIME)))
675  {
676  overlay_.peerFinder().on_failure(slot_);
677  fail("Not useful");
678  return;
679  }
680  }
681 
682  // Already waiting for PONG
683  if (lastPingSeq_)
684  {
685  fail("Ping Timeout");
686  return;
687  }
688 
689  lastPingTime_ = clock_type::now();
690  lastPingSeq_ = rand_int<std::uint32_t>();
691 
692  protocol::TMPing message;
693  message.set_type(protocol::TMPing::ptPING);
694  message.set_seq(*lastPingSeq_);
695 
696  send(std::make_shared<Message>(message, protocol::mtPING));
697 
698  setTimer();
699 }

716 //------------------------------------------------------------------------------
717 void
718 PeerImp::doAccept()
719 {
720  assert(read_buffer_.size() == 0);
721 
722  JLOG(journal_.debug()) << "doAccept: " << remote_address_;
723 
724  auto const sharedValue = makeSharedValue(*stream_ptr_, journal_);
725 
726  // This shouldn't fail since we already computed
727  // the shared value successfully in OverlayImpl
728  if (!sharedValue)
729  return fail("makeSharedValue: Unexpected failure");
730 
731  JLOG(journal_.info()) << "Protocol: " << to_string(protocol_);
732  JLOG(journal_.info()) << "Public Key: "
733  << toBase58(TokenType::NodePublic, publicKey_);
734 
735  if (auto member = app_.cluster().member(publicKey_))
736  {
737  {
738  std::unique_lock lock{nameMutex_};
739  name_ = *member;
740  }
741  JLOG(journal_.info()) << "Cluster name: " << *member;
742  }
743 
744  overlay_.activate(shared_from_this());
745 
746  // XXX Set timer: connection is in grace period to be useful.
747  // XXX Set timer: connection idle (idle may vary depending on connection
748  // type.)
749 
750  auto write_buffer = std::make_shared<boost::beast::multi_buffer>();
751 
752  boost::beast::ostream(*write_buffer) << makeResponse(
753  !overlay_.peerFinder().config().peerPrivate,
754  request_,
755  overlay_.setup().public_ip,
756  remote_address_.address(),
757  *sharedValue,
758  overlay_.setup().networkID,
759  protocol_,
760  app_);
761 
762  // Write the whole buffer and only start protocol when that's done.
763  boost::asio::async_write(
764  stream_,
765  write_buffer->data(),
766  boost::asio::transfer_all(),
767  bind_executor(
768  strand_,
769  [this, write_buffer, self = shared_from_this()](
770  error_code ec, std::size_t bytes_transferred) {
