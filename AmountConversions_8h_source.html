78 {
79  assert(amt.mantissa() < std::numeric_limits<std::int64_t>::max());
80  bool const isNeg = amt.negative();
81  std::int64_t const sMant =
82  isNeg ? -std::int64_t(amt.mantissa()) : amt.mantissa();

84  assert(!isXRP(amt));
