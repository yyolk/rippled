38 // account: account,
39 // ledger_index_min: ledger_index,
40 // ledger_index_max: ledger_index,
41 // binary: boolean, // optional, defaults to false
42 // count: boolean, // optional, defaults to false

44 // offset: integer, // optional, defaults to 0
45 // limit: integer // optional

47 Json::Value
48 doAccountTxOld(RPC::JsonContext& context)
49 {
50  std::uint32_t offset = context.params.isMember(jss::offset)
51  ? context.params[jss::offset].asUInt()
52  : 0;
53  std::uint32_t limit = context.params.isMember(jss::limit)
54  ? context.params[jss::limit].asUInt()
55  : UINT32_MAX;
56  bool bBinary = context.params.isMember(jss::binary) &&
57  context.params[jss::binary].asBool();
58  bool bDescending = context.params.isMember(jss::descending) &&
59  context.params[jss::descending].asBool();
60  bool bCount = context.params.isMember(jss::count) &&
61  context.params[jss::count].asBool();
62  std::uint32_t uLedgerMin;
63  std::uint32_t uLedgerMax;
64  std::uint32_t uValidatedMin;
65  std::uint32_t uValidatedMax;
66  bool bValidated =
67  context.ledgerMaster.getValidatedRange(uValidatedMin, uValidatedMax);
68 
69  if (!context.params.isMember(jss::account))
70  return rpcError(rpcINVALID_PARAMS);
71 
72  auto const raAccount =
73  parseBase58<AccountID>(context.params[jss::account].asString());
74  if (!raAccount)
75  return rpcError(rpcACT_MALFORMED);
76 
77  if (offset > 3000)
78  return rpcError(rpcATX_DEPRECATED);
79 
80  context.loadType = Resource::feeHighBurdenRPC;
81 
82  // DEPRECATED
83  if (context.params.isMember(jss::ledger_min))
84  {
85  context.params[jss::ledger_index_min] = context.params[jss::ledger_min];
86  bDescending = true;
87  }
88 
89  // DEPRECATED
90  if (context.params.isMember(jss::ledger_max))
91  {
92  context.params[jss::ledger_index_max] = context.params[jss::ledger_max];
93  bDescending = true;
94  }
95 
96  if (context.params.isMember(jss::ledger_index_min) ||
97  context.params.isMember(jss::ledger_index_max))
98  {
99  std::int64_t iLedgerMin = context.params.isMember(jss::ledger_index_min)
100  ? context.params[jss::ledger_index_min].asInt()
101  : -1;
102  std::int64_t iLedgerMax = context.params.isMember(jss::ledger_index_max)
103  ? context.params[jss::ledger_index_max].asInt()
104  : -1;
105 
106  if (!bValidated && (iLedgerMin == -1 || iLedgerMax == -1))
107  {
108  // Don't have a validated ledger range.
109  if (context.apiVersion == 1)
110  return rpcError(rpcLGR_IDXS_INVALID);
111  return rpcError(rpcNOT_SYNCED);
112  }
113 
114  uLedgerMin = iLedgerMin == -1 ? uValidatedMin : iLedgerMin;
115  uLedgerMax = iLedgerMax == -1 ? uValidatedMax : iLedgerMax;
116 
117  if (uLedgerMax < uLedgerMin)
118  {
119  if (context.apiVersion == 1)
120  return rpcError(rpcLGR_IDXS_INVALID);
121  return rpcError(rpcNOT_SYNCED);
122  }
123  }
124  else
125  {
126  std::shared_ptr<ReadView const> ledger;
127  auto ret = RPC::lookupLedger(ledger, context);
128 
129  if (!ledger)
130  return ret;
131 
132  if (!ret[jss::validated].asBool() ||
133  (ledger->info().seq > uValidatedMax) ||
134  (ledger->info().seq < uValidatedMin))
135  {
136  return rpcError(rpcLGR_NOT_VALIDATED);
137  }
138 
139  uLedgerMin = uLedgerMax = ledger->info().seq;
140  }
141 
142  int count = 0;
143 
144 #ifndef DEBUG
145 
146  try
147  {
148 #endif
149 
150  Json::Value ret(Json::objectValue);
151 
152  ret[jss::account] = context.app.accountIDCache().toBase58(*raAccount);
153  Json::Value& jvTxns = (ret[jss::transactions] = Json::arrayValue);
154 
155  RelationalDBInterface::AccountTxOptions options = {
156  *raAccount,
157  uLedgerMin,
158  uLedgerMax,
159  offset,
160  limit,
161  isUnlimited(context.role)};
162 
163  if (bBinary)
164  {
165  std::vector<RelationalDBInterface::txnMetaLedgerType> txns;
166 
167  if (bDescending)
168  txns = dynamic_cast<RelationalDBInterfaceSqlite*>(
169  &context.app.getRelationalDBInterface())
170  ->getNewestAccountTxsB(options);
171  else
172  txns = dynamic_cast<RelationalDBInterfaceSqlite*>(
173  &context.app.getRelationalDBInterface())
174  ->getOldestAccountTxsB(options);
175 
176  for (auto it = txns.begin(), end = txns.end(); it != end; ++it)
177  {
178  ++count;
179  Json::Value& jvObj = jvTxns.append(Json::objectValue);
180 
181  std::uint32_t uLedgerIndex = std::get<2>(*it);
182  jvObj[jss::tx_blob] = strHex(std::get<0>(*it));
183  jvObj[jss::meta] = strHex(std::get<1>(*it));
184  jvObj[jss::ledger_index] = uLedgerIndex;
185  jvObj[jss::validated] = bValidated &&
186  uValidatedMin <= uLedgerIndex &&
187  uValidatedMax >= uLedgerIndex;
188  }
189  }
190  else
191  {
192  RelationalDBInterface::AccountTxs txns;
193 
194  if (bDescending)
195  txns = dynamic_cast<RelationalDBInterfaceSqlite*>(
196  &context.app.getRelationalDBInterface())
197  ->getNewestAccountTxs(options);
198  else
199  txns = dynamic_cast<RelationalDBInterfaceSqlite*>(
200  &context.app.getRelationalDBInterface())
201  ->getOldestAccountTxs(options);
202 
203  for (auto it = txns.begin(), end = txns.end(); it != end; ++it)
204  {
205  ++count;
206  Json::Value& jvObj = jvTxns.append(Json::objectValue);
207 
208  if (it->first)
209  jvObj[jss::tx] =
210  it->first->getJson(JsonOptions::include_date);
211 
212  if (it->second)
213  {
214  std::uint32_t uLedgerIndex = it->second->getLgrSeq();
215 
216  auto meta = it->second->getJson(JsonOptions::none);
217  insertDeliveredAmount(
218  meta, context, it->first, *it->second);
219  jvObj[jss::meta] = std::move(meta);
220 
221  jvObj[jss::validated] = bValidated &&
222  uValidatedMin <= uLedgerIndex &&
223  uValidatedMax >= uLedgerIndex;
224  }
225  }
226  }
227 
228  // Add information about the original query
229  ret[jss::ledger_index_min] = uLedgerMin;
230  ret[jss::ledger_index_max] = uLedgerMax;
231  ret[jss::validated] = bValidated && uValidatedMin <= uLedgerMin &&
232  uValidatedMax >= uLedgerMax;
233  ret[jss::offset] = offset;
234 
235  // We no longer return the full count but only the count of returned
236  // transactions. Computing this count was two expensive and this API is
237  // deprecated anyway.
238  if (bCount)
239  ret[jss::count] = count;
240 
241  if (context.params.isMember(jss::limit))
242  ret[jss::limit] = limit;
243 
244  return ret;
245 #ifndef DEBUG
246  }
247  catch (std::exception const&)
248  {
249  return rpcError(rpcINTERNAL);
250  }
251 
252 #endif
253 }
