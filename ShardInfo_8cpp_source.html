75 protocol::TMPeerShardInfoV2
76 ShardInfo::makeMessage(Application& app)
77 {
78  protocol::TMPeerShardInfoV2 message;
79  Serializer s;
80  s.add32(HashPrefix::shardInfo);
81 
82  // Set the message creation time
83  msgTimestamp_ = app.timeKeeper().now();
84  {
85  auto const timestamp{msgTimestamp_.time_since_epoch().count()};
86  message.set_timestamp(timestamp);
87  s.add32(timestamp);
88  }
89 
90  if (!incomplete_.empty())
91  {
92  message.mutable_incomplete()->Reserve(incomplete_.size());
93  for (auto const& [shardIndex, incomplete] : incomplete_)
94  {
95  auto tmIncomplete{message.add_incomplete()};
96 
97  tmIncomplete->set_shardindex(shardIndex);
98  s.add32(shardIndex);
99 
100  static_assert(std::is_same_v<
101  std::underlying_type_t<decltype(incomplete.state())>,
102  std::uint32_t>);
103  auto const state{static_cast<std::uint32_t>(incomplete.state())};
104  tmIncomplete->set_state(state);
105  s.add32(state);
106 
107  // Set progress if greater than zero
108  auto const percentProgress{incomplete.percentProgress()};
109  if (percentProgress > 0)
110  {
111  tmIncomplete->set_progress(percentProgress);
112  s.add32(percentProgress);
113  }
114  }
115  }
116 
117  if (!finalized_.empty())
118  {
119  auto const str{ripple::to_string(finalized_)};
120  message.set_finalized(str);
121  s.addRaw(str.data(), str.size());
122  }
123 
124  // Set the public key
125  auto const& publicKey{app.nodeIdentity().first};
126  message.set_publickey(publicKey.data(), publicKey.size());
127 
128  // Create a digital signature using the node private key
129  auto const signature{sign(publicKey, app.nodeIdentity().second, s.slice())};
130 
131  // Set the digital signature
132  message.set_signature(signature.data(), signature.size());
133 
134  return message;
135 }
