30  STObject const& obj,

32  std::string const& annotation)
33 {
34  std::pair<std::vector<SignerEntry>, NotTEC> s;
35 
36  if (!obj.isFieldPresent(sfSignerEntries))
37  {
38  JLOG(journal.trace())
39  << "Malformed " << annotation << ": Need signer entry array.";
40  s.second = temMALFORMED;
41  return s;
42  }
43 
44  auto& accountVec = s.first;
45  accountVec.reserve(STTx::maxMultiSigners);
46 
47  STArray const& sEntries(obj.getFieldArray(sfSignerEntries));
48  for (STObject const& sEntry : sEntries)
49  {
50  // Validate the SignerEntry.
51  if (sEntry.getFName() != sfSignerEntry)
52  {
53  JLOG(journal.trace())
54  << "Malformed " << annotation << ": Expected SignerEntry.";
55  s.second = temMALFORMED;
56  return s;
57  }
58 
59  // Extract SignerEntry fields.
60  AccountID const account = sEntry.getAccountID(sfAccount);
61  std::uint16_t const weight = sEntry.getFieldU16(sfSignerWeight);
62  accountVec.emplace_back(account, weight);
63  }
64 
65  s.second = tesSUCCESS;
66  return s;
67 }
