69 TimeoutCounter::queueJob(ScopedLockType& sl)
70 {
71  if (isDone())
72  return;
73  if (queueJobParameter_.jobLimit &&
74  app_.getJobQueue().getJobCountTotal(queueJobParameter_.jobType) >=
75  queueJobParameter_.jobLimit)
76  {
77  JLOG(journal_.debug()) << "Deferring " << queueJobParameter_.jobName
78  << " timer due to load";
79  setTimer(sl);
80  return;
81  }
82 
83  app_.getJobQueue().addJob(
84  queueJobParameter_.jobType,
85  queueJobParameter_.jobName,
86  [wptr = pmDowncast()](Job&) {
87  if (auto sptr = wptr.lock(); sptr)
88  sptr->invokeOnTimer();
89  });
90 }

92 void
93 TimeoutCounter::invokeOnTimer()
94 {
95  ScopedLockType sl(mtx_);
96 
97  if (isDone())
98  return;
99 
100  if (!progress_)
101  {
102  ++timeouts_;
103  JLOG(journal_.debug()) << "Timeout(" << timeouts_ << ") "
104  << " acquiring " << hash_;
105  onTimer(false, sl);
106  }
107  else
108  {
109  progress_ = false;
110  onTimer(true, sl);
111  }
112 
113  if (!isDone())
114  setTimer(sl);
115 }
