108  T
109  percentile(float p) const
110  {
111  assert(p >= 0 && p <= 1);
112  std::size_t pos = std::round(p * samples);
113 
114  if (counts_.empty())
115  return T{};
116 
117  auto it = counts_.begin();
118  std::size_t cumsum = it->second;
119  while (it != counts_.end() && cumsum < pos)
120  {
121  ++it;
122  cumsum += it->second;
123  }
124  return it->first;
125  }
