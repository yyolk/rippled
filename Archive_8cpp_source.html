28 void
29 extractTarLz4(
30  boost::filesystem::path const& src,
31  boost::filesystem::path const& dst)
32 {
33  if (!is_regular_file(src))
34  Throw<std::runtime_error>("Invalid source file");
35 
36  using archive_ptr =
37  std::unique_ptr<struct archive, void (*)(struct archive*)>;
38  archive_ptr ar{
39  archive_read_new(), [](struct archive* a) { archive_read_free(a); }};
40  if (!ar)
41  Throw<std::runtime_error>("Failed to allocate archive");
42 
43  if (archive_read_support_format_tar(ar.get()) < ARCHIVE_OK)
44  Throw<std::runtime_error>(archive_error_string(ar.get()));
45 
46  if (archive_read_support_filter_lz4(ar.get()) < ARCHIVE_OK)
47  Throw<std::runtime_error>(archive_error_string(ar.get()));
48 
49  // Examples suggest this block size
50  if (archive_read_open_filename(ar.get(), src.string().c_str(), 10240) <
51  ARCHIVE_OK)
52  {
53  Throw<std::runtime_error>(archive_error_string(ar.get()));
54  }
55 
56  archive_ptr aw{archive_write_disk_new(), [](struct archive* a) {
57  archive_write_free(a);
58  }};
59  if (!aw)
60  Throw<std::runtime_error>("Failed to allocate archive");
61 
62  if (archive_write_disk_set_options(
63  aw.get(),
64  ARCHIVE_EXTRACT_TIME | ARCHIVE_EXTRACT_PERM | ARCHIVE_EXTRACT_ACL |
65  ARCHIVE_EXTRACT_FFLAGS) < ARCHIVE_OK)
66  {
67  Throw<std::runtime_error>(archive_error_string(aw.get()));
68  }
69 
70  if (archive_write_disk_set_standard_lookup(aw.get()) < ARCHIVE_OK)
71  Throw<std::runtime_error>(archive_error_string(aw.get()));
72 
73  int result;
74  struct archive_entry* entry;
75  while (true)
76  {
77  result = archive_read_next_header(ar.get(), &entry);
78  if (result == ARCHIVE_EOF)
79  break;
80  if (result < ARCHIVE_OK)
81  Throw<std::runtime_error>(archive_error_string(ar.get()));
82 
83  archive_entry_set_pathname(
84  entry, (dst / archive_entry_pathname(entry)).string().c_str());
85  if (archive_write_header(aw.get(), entry) < ARCHIVE_OK)
86  Throw<std::runtime_error>(archive_error_string(aw.get()));
87 
88  if (archive_entry_size(entry) > 0)
89  {
90  const void* buf;
91  size_t sz;
92  la_int64_t offset;
93  while (true)
94  {
95  result = archive_read_data_block(ar.get(), &buf, &sz, &offset);
96  if (result == ARCHIVE_EOF)
97  break;
98  if (result < ARCHIVE_OK)
99  Throw<std::runtime_error>(archive_error_string(ar.get()));
100 
101  if (archive_write_data_block(aw.get(), buf, sz, offset) <
102  ARCHIVE_OK)
103  {
104  Throw<std::runtime_error>(archive_error_string(aw.get()));
105  }
106  }
107  }
108 
109  if (archive_write_finish_entry(aw.get()) < ARCHIVE_OK)
110  Throw<std::runtime_error>(archive_error_string(aw.get()));
111  }
112 }
