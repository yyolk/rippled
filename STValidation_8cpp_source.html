88 bool
89 STValidation::isValid() const noexcept
90 {
91  if (!valid_)
92  {
93  assert(publicKeyType(getSignerPublic()) == KeyType::secp256k1);
94 
95  valid_ = verifyDigest(
96  getSignerPublic(),
97  getSigningHash(),
98  makeSlice(getFieldVL(sfSignature)),
99  getFlags() & vfFullyCanonicalSig);
100  }
101 
102  return valid_.value();
103 }
